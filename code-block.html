" GRIDPOINTS ARRAY FUNCTIONS "
gpos = gpa.pos()
gx, gy, gz = gpos.T
print gz
f = gpa.fixity()
print f
f[:,][gz==0] = True, True, True
print f
gpa.set_fixity(f)

top_gridpoints = gz==10
radial_distance = np.sqrt((gx-5)**2+(gy-5)**2)
central_gridpoints = radial_distance < 5
mask = np.logical_and(top_gridpoints, central_gridpoints)
print "boundary load applied to {} gridpoints".format(mask.sum())
fapp = gpa.force_app()
print fapp
fapp[:,2] = mask*1e6*(5.0-radial_distance)/5.0
gpa.set_force_app(fapp)

print za.pos()
za.gridpoints()
za.faces()
za.ids()
print za.neighbors()

6.2 KIAST Model 

import itasca as it
import numpy as np
np.set_printoptions(threshold=20)
it.command("python-reset-state false")
from itasca import zonearray as za
from itasca import gridpointarray as gpa

command_template_zone = """
model new
zone create radial-cylinder point 0 (0,0,0) ...
                            point 1 (40,0,0) ...
                            point 2 (0,10,0) ...
                            point 3 (0,0,40) ...
                            point 4 (40,10,0) ...
                            point 5 (0,10,40) ...
                            point 6 (40,0,40) ...
                            point 7 (40,10,40) ...
                            point 8 (1,0,0) ...
                            point 9 (0,0,1) ...
                            point 10 (1,10,0) ...
                            point 11 (0,10,1) ...
                            size {radial} {axial} {perimeter} {outer} ...
                            rat 1 1 1 1.5 ...
                            fill group "shaft"
"""

_radial = 10
_axial = 20
_perimeter = 26 
_outer = 40

command = command_template_zone.format(radial=_radial, axial = _axial, perimeter = _perimeter, outer = _outer)
it.command(command)


it.command("""
zone group "plate" range position-x 0 3.2 position-y 8.5 10 position-z 0 3.2
""")

it.command("""
zone cmodel assign strain-softening range group "Radial Cylinder1"
zone property density 2500 bulk 2e8 shear 1e8 range group "Radial Cylinder1"
zone property cohesion 2e6 friction 45 tension 2e5 dilation 10 range group "Radial Cylinder1"
zone property table-friction 'fri' table-cohesion 'coh' table-dilation 'dil' range group "Radial Cylinder1"
table 'fri' add (0, 45) (.05, 42) (.1, 40) (1, 40)
table 'coh' add (0,2e6) (.05,1e6) (.1,5e5) (1,5e5) 
table 'dil' add (0, 10) (.05,  3) (.1,  0) 
;Constitutive modeland properties for shaft and plate
zone cmodel assign elastic range group 'shaft'
zone property bulk {name}  shear 10.4e9  range group 'shaft'
zone cmodel assign elastic range group 'plate'
zone property bulk {name}  shear 10.4e9  range group 'plate'
""".format(name=13.9e9))


p = za.pos()
type(p)
p.shape
it.zone.count()
x,y,z = p.T
upper_zones = z > 1.5
print((upper_zones).sum())

modulus = np.zeros_like(x) #create array same shape as x
modulus[upper_zones] = 1.2e10#specify all upper zones have modulus 1.2e10
lower_zones = np.logical_not(upper_zones)
modulus[lower_zones] = 1.6e10
za.set_prop_scalar("young", modulus)

it.command("zone group \"lower\" range position-z 0 5")
za.in_group("lower")
za.in_group("lower").sum() #how many zones within specific group

#refine this by getting corners
#create a new mask
corner_mask = reduce(np.logical_and, (x<3, y<3, z<3)) #reduce is simply applying logical and x<3, y<3, z<3
za.set_group(corner_mask, "corner", "geometry") # set the zones with corner_mask = true have "corner" and "geometry"
print(za.in_group("corner", "geometry").sum(), "zones in corner group.") #output how many zones are in the corner group
