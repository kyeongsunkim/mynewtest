1 KAIST Model FLAC3D

1.1 Grid Generation 

import itasca as it
import numpy as np
np.set_printoptions(threshold=20)
it.command("python-reset-state false")
from itasca import zonearray as za
from itasca import gridpointarray as gpa
" =========================================== "
" =============GRID GENERATION=============== "
" =========================================== "
#
it.command("""
model new
zone create radial-cylinder point 0 (0,0,0) ...
                            point 1 (40,0,0) ...
                            point 2 (0,10,0) ...
                            point 3 (0,0,40) ...
                            point 4 (40,10,0) ...
                            point 5 (0,10,40) ...
                            point 6 (40,0,40) ...
                            point 7 (40,10,40) ...
                            point 8 (1,0,0) ...
                            point 9 (0,0,1) ...
                            point 10 (1,10,0) ...
                            point 11 (0,10,1) ...
                            size 10 20 26 40 ...
                            rat 1 1 1 1.5 ...
                            fill group "shaft"
          """)
